def read_cmudict_block(stream):
    entries = []
    while len(entries) < 100: // Read 100 at a time.
        line = stream.readline()
        if line == "": return entries // end of file.
        pieces = line.split()
        entries.append( (pieces[0], int(pieces[1]), tuple(pieces[2:])) )
    return entries
